        // App Configuration
        const APPS = [
            { id: 1, name: 'Nature', icon: 'üçÉ', color: 'teal', defaultProgress: 0 },
            { id: 2, name: 'The Lab', icon: 'üß†', color: 'teal', defaultProgress: 0 },
            { id: 3, name: 'The Podium', icon: 'üçé', color: 'teal', defaultProgress: 0 },
            { id: 4, name: 'The Hub', icon: 'ü§ù', color: 'teal', defaultProgress: 0 },
            { id: 5, name: 'Life', icon: 'ü™∑', color: 'teal', defaultProgress: 0 },
            { id: 6, name: 'Vault', icon: 'üíé', color: 'teal', defaultProgress: 0 },
            { id: 7, name: 'Pulse', icon: 'üèãÔ∏è‚Äç‚ôÄÔ∏è', color: 'pink', defaultProgress: 0 },
            { id: 8, name: 'Dev', icon: '‚¨°', color: 'teal', defaultProgress: 0 }
        ];

        // State Management
        let currentView = 'zen';
        let currentActivatingApp = null;
        let appsData = {};

        // Initialize
        document.addEventListener('DOMContentLoaded', () => {
            loadAppsData();
            renderApps();
            updateClock();
            setInterval(updateClock, 1000);
            setupEventListeners();
            updateSystemsOnline();
        });

        // Load Apps Data from localStorage
        function loadAppsData() {
            const stored = localStorage.getItem('missionControlApps');
            if (stored) {
                appsData = JSON.parse(stored);
            } else {
                // Initialize with empty data
                APPS.forEach(app => {
                    appsData[app.id] = {
                        url: null,
                        progress: app.defaultProgress,
                        active: false
                    };
                });
                saveAppsData();
            }
        }

        // Save Apps Data to localStorage
        function saveAppsData() {
            localStorage.setItem('missionControlApps', JSON.stringify(appsData));
        }

        // Render Apps
        function renderApps() {
            const library = document.getElementById('appLibrary');
            library.innerHTML = '';

            APPS.forEach(app => {
                const appData = appsData[app.id];
                const isActive = appData && appData.active;
                
                const card = document.createElement('div');
                card.className = `app-card ${!isActive ? 'inactive' : ''}`;
                card.setAttribute('data-app-id', app.id);

                const progress = appData?.progress || 0;
                const circumference = 2 * Math.PI * 77;
                const offset = circumference - (progress / 100) * circumference;

                card.innerHTML = `
                    <div class="app-card-inner">
                        <div class="app-card-front">
                            <svg class="progress-ring ${app.color}">
                                <circle class="bg" cx="80" cy="80" r="77"></circle>
                                <circle 
                                    class="progress" 
                                    cx="80" 
                                    cy="80" 
                                    r="77"
                                    stroke-dasharray="${circumference}"
                                    stroke-dashoffset="${offset}"
                                ></circle>
                            </svg>
                            <div class="app-icon">${app.icon}</div>
                            <div class="app-name">${app.name}</div>
                        </div>
                        <div class="app-card-back">
                            <h3 style="font-size: 16px; margin-bottom: 16px;">${app.name}</h3>
                            <input 
                                type="url" 
                                class="edit-input" 
                                placeholder="GitHub URL..."
                                value="${appData?.url || ''}"
                            >
                            <div style="display: flex; gap: 8px; margin-top: 12px;">
                                <button class="edit-button" onclick="saveAppUrl(${app.id})">Save</button>
                                <button class="reset-button" onclick="resetApp(${app.id})">Reset</button>
                            </div>
                        </div>
                    </div>
                `;

                // Single click - open app or activate
                card.addEventListener('click', (e) => {
                    if (!card.classList.contains('flipped')) {
                        if (isActive) {
                            openApp(app.id);
                        } else {
                            showActivationView(app.id);
                        }
                    }
                });

                // Double click - flip card
                card.addEventListener('dblclick', (e) => {
                    e.stopPropagation();
                    card.classList.toggle('flipped');
                });

                library.appendChild(card);
            });
        }

        // Show Activation View
        function showActivationView(appId) {
            const app = APPS.find(a => a.id === appId);
            currentActivatingApp = appId;
            
            document.getElementById('activationAppName').textContent = app.name.toUpperCase();
            document.getElementById('activationInput').value = '';
            document.getElementById('activationView').classList.add('active');
        }

        // Activate App
        document.getElementById('activateButton')?.addEventListener('click', () => {
            const url = document.getElementById('activationInput').value.trim();
            
            if (!url) {
                alert('Please enter a valid URL');
                return;
            }

            if (!url.startsWith('http://') && !url.startsWith('https://')) {
                alert('URL must start with http:// or https://');
                return;
            }

            appsData[currentActivatingApp] = {
                url: url,
                progress: 0,
                active: true
            };

            saveAppsData();
            document.getElementById('activationView').classList.remove('active');
            renderApps();
            updateSystemsOnline();
        });

        // Close Activation View
        document.getElementById('closeActivation')?.addEventListener('click', () => {
            document.getElementById('activationView').classList.remove('active');
        });

        // Open App in Iframe
        function openApp(appId) {
            const appData = appsData[appId];
            if (!appData || !appData.url) return;

            document.getElementById('appIframe').src = appData.url;
            document.getElementById('appIframeView').classList.add('active');
        }

        // Close App Iframe
        document.getElementById('doneButton')?.addEventListener('click', () => {
            document.getElementById('appIframeView').classList.remove('active');
            document.getElementById('appIframe').src = '';
        });

        // Save App URL from Edit View
        function saveAppUrl(appId) {
            const card = document.querySelector(`[data-app-id="${appId}"]`);
            const input = card.querySelector('.edit-input');
            const url = input.value.trim();

            if (url && !url.startsWith('http://') && !url.startsWith('https://')) {
                alert('URL must start with http:// or https://');
                return;
            }

            appsData[appId].url = url;
            appsData[appId].active = !!url;
            saveAppsData();
            
            card.classList.toggle('flipped');
            renderApps();
            updateSystemsOnline();
        }

        // Reset App
        function resetApp(appId) {
            if (!confirm('Are you sure you want to reset this app?')) return;

            appsData[appId] = {
                url: null,
                progress: 0,
                active: false
            };

            saveAppsData();
            
            const card = document.querySelector(`[data-app-id="${appId}"]`);
            card.classList.toggle('flipped');
            renderApps();
            updateSystemsOnline();
        }

        // Update Systems Online Counter
        function updateSystemsOnline() {
            const activeCount = Object.values(appsData).filter(app => app.active).length;
            document.getElementById('systemsOnline').textContent = `${activeCount}/8 Online`;
        }

        // Command Input Handler
        const commandInput = document.getElementById('commandInput');
        commandInput?.addEventListener('keydown', (e) => {
            if (e.key === 'Enter') {
                handleCommand(e.target.value.trim().toLowerCase());
                e.target.value = '';
            }
        });

        // Handle Commands
        function handleCommand(command) {
            if (command === 'apps') {
                switchToLibrary();
            } else if (command === 'clear') {
                switchToZen();
            } else {
                // Check if command matches an app name
                const app = APPS.find(a => a.name.toLowerCase() === command);
                if (app) {
                    switchToLibrary();
                    setTimeout(() => {
                        const appData = appsData[app.id];
                        if (appData && appData.active) {
                            openApp(app.id);
                        } else {
                            showActivationView(app.id);
                        }
                    }, 300);
                }
            }
        }

        // Switch Views
        function switchToLibrary() {
            currentView = 'library';
            document.querySelector('.greeting-pill').style.display = 'none';
            document.querySelector('.command-bar').style.display = 'none';
            document.querySelector('.clock').style.display = 'none';
            document.querySelector('.date').style.display = 'none';
            document.getElementById('appLibrary').classList.add('active');
            document.getElementById('bottomNav').classList.add('active');
            document.getElementById('syncButton').classList.add('active');
        }

        function switchToZen() {
            currentView = 'zen';
            document.querySelector('.greeting-pill').style.display = 'flex';
            document.querySelector('.command-bar').style.display = 'flex';
            document.querySelector('.clock').style.display = 'block';
            document.querySelector('.date').style.display = 'block';
            document.getElementById('appLibrary').classList.remove('active');
            document.getElementById('bottomNav').classList.remove('active');
            document.getElementById('syncButton').classList.remove('active');
        }

        // Navigation Items
        document.querySelectorAll('.nav-item').forEach(item => {
            item.addEventListener('click', () => {
                const view = item.getAttribute('data-view');
                
                document.querySelectorAll('.nav-item').forEach(i => i.classList.remove('active'));
                item.classList.add('active');

                if (view === 'zen') {
                    switchToZen();
                } else if (view === 'library') {
                    switchToLibrary();
                }
            });
        });

        // Sync to GitHub
        document.getElementById('syncButton')?.addEventListener('click', () => {
            const config = {
                timestamp: new Date().toISOString(),
                apps: appsData
            };

            const blob = new Blob([JSON.stringify(config, null, 2)], { type: 'application/json' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'mission-control-config.json';
            a.click();
            URL.revokeObjectURL(url);

            alert('Configuration downloaded! Commit this to your GitHub repository.');
        });

        // Update Clock
        function updateClock() {
            const now = new Date();
            const hours = String(now.getHours()).padStart(2, '0');
            const minutes = String(now.getMinutes()).padStart(2, '0');
            document.getElementById('clock').textContent = `${hours}:${minutes}`;

            const days = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'];
            const months = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December'];
            const dayName = days[now.getDay()];
            const monthName = months[now.getMonth()];
            const date = now.getDate();
            
            document.getElementById('date').textContent = `${dayName.toUpperCase()}, ${monthName.toUpperCase()} ${date}`;

            // Update greeting based on time
            const hour = now.getHours();
            let greeting = 'Greetings';
            if (hour < 12) greeting = 'Good Morning';
            else if (hour < 18) greeting = 'Good Afternoon';
            else greeting = 'Good Evening';
            
            document.getElementById('greetingText').textContent = `${greeting}, User`;
        }

        // Setup Event Listeners
        function setupEventListeners() {
            // Escape key to close views
            document.addEventListener('keydown', (e) => {
                if (e.key === 'Escape') {
                    if (document.getElementById('appIframeView').classList.contains('active')) {
                        document.getElementById('doneButton').click();
                    } else if (document.getElementById('activationView').classList.contains('active')) {
                        document.getElementById('closeActivation').click();
                    }
                }
            });
        }

        // Make functions global for inline handlers
        window.saveAppUrl = saveAppUrl;
        window.resetApp = resetApp;
    </script>
